\begin{theorem}
Assume FO^k-admissible refinement with
\[
\forall t,\; I(X;\mathcal{T}_{t+1}\mid \mathcal{T}_t) \le \frac{C}{\log n},
\]
and exact recovery $H(X\mid \mathcal{T}_T)=0$. Then
\[
T \ge \frac{\log n}{C} H(X).
\]
In particular, if $H(X) \ge c_0 n$, then
\[
T = \Omega(n\log n).
\]
\end{theorem}


% Graph-structured superlinear extension
\begin{theorem}
Let $X$ range over graphs on $n$ vertices. Then for $k \ge 3$,
\[
\log N_k(n) = \Omega(n^2),
\]
and any FO^k-admissible refinement process satisfies
\[
T = \Omega(n^2).
\]
\end{theorem}


% Correction: FO^k class size bound
\begin{lemma}
For fixed $k$, FO^k-equivalence classes of graphs satisfy
\[
|\{G' : G' \equiv_k G\}| \le 2^{C_k n \log n}.
\]
\end{lemma}

\begin{theorem}
For graphs,
\[
\log N_k(n) = \Theta(n^2),
\quad
T = \Omega(n^2).
\]
\end{theorem}


% Correction: SAT limitation of FO^k framework
\begin{proposition}
For $k$-CNF with $m = \alpha n$ or $m = n^{1+\varepsilon}$, FO^k-equivalence may collapse exponentially many instances:
\[
|\{F' : F' \equiv_k F\}| \ge 2^{\Omega(m)}.
\]
Hence
\[
\log N_k(\mathrm{SAT}_{n,m}) \not\ge \Omega(m),
\]
and no superlinear lower bound follows from FO^k-type counting.
\end{proposition}

\begin{remark}
The entropy-depth framework yields tight bounds for linear orders and graphs but is insufficient for SAT. Lower bounds in SAT require non-local methods such as resolution complexity or switching lemmas.
\end{remark}


% Blocking Frontier: Non-local proof systems
\section{Blocking Frontier: Non-Local Methods}

\begin{definition}[Non-Local Proof System Integration]
A refinement framework is said to integrate non-local methods if it can simulate or encode proof systems whose inference steps depend on global structure rather than bounded FO$^k$-local neighborhoods.
\end{definition}

\begin{problem}[Blocking Frontier]
Extend the entropy-depth framework to incorporate non-local proof systems such as:
\begin{itemize}
\item Resolution (width/size tradeoffs)
\item Switching lemma methods (random restrictions)
\end{itemize}
and derive lower bounds of the form:
\[
T(n) \ge \Omega(n^{1+\varepsilon})
\]
for SAT in the regime $m = \alpha n$.
\end{problem}

\begin{obstruction}
FO$^k$-local refinement cannot distinguish instances beyond bounded Gaifman radius, leading to collapse:
\[
|\{F' : F' \equiv_k F\}| \ge 2^{\Omega(m)}.
\]
Thus any extension must break locality.
\end{obstruction}

\begin{goal}
Construct a hybrid model where:
\[
\text{Local information flow} + \text{Non-local inference} \;\Rightarrow\; \text{superlinear lower bounds}.
\]
\end{goal}
