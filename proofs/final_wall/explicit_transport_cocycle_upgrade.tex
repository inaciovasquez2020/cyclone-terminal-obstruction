\begin{definition}[Edge-flip transport cocycle]
For adjacent $u\sim v$, define
\[
\tau_{uv} = \sum_{e\in E(u)\cap E(v)} (\sigma_u(e)+\sigma_v(e)) \in \mathbb F_2.
\]
\end{definition}

\begin{lemma}[Cocycle identity]
\[
\tau_{uv}+\tau_{vw}+\tau_{wu}=0.
\]
\end{lemma}
