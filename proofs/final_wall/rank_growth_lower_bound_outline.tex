\section*{Rank-Growth Lower Bound Outline}

\[
\textbf{Goal:}\quad C_R(G)\ge T \Rightarrow |\mathrm{Types}_{FO^k_R}(G)|\ge 2.
\]

\[
\textbf{Step 1:}\quad
C_R(G)=\operatorname{rank}_{\mathbb F_2}(M_R(G)),
\]
where \(M_R(G)\) is the radius-\(R\) overlap form on \(H_1(G;\mathbb F_2)\).

\[
\textbf{Step 2:}\quad
|\mathrm{Types}_{FO^k_R}(G)|=1
\Rightarrow
\text{all rooted radius-}R\text{ neighborhoods are }FO^k\text{-equivalent.}
\]

\[
\textbf{Step 3:}\quad
|\mathrm{Types}_{FO^k_R}(G)|=1
\Rightarrow
\operatorname{rank}(M_R(G))\le B(k,\Delta,R).
\]

\[
\textbf{Step 4:}\quad
C_R(G)>B(k,\Delta,R)
\Rightarrow
|\mathrm{Types}_{FO^k_R}(G)|\ge 2.
\]

\[
\textbf{Missing lemma:}\quad
\text{uniform bounded-type homogeneity forces bounded overlap rank.}
\]
