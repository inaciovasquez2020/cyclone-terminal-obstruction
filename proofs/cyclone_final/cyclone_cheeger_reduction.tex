\section*{Cyclone Reduction via Cheeger Inequality}

\[
\lambda_1(B)\ge \frac{h(B)^2}{2\Delta}
\]

\[
h(B)\ge \frac{c(d)}{\operatorname{diam}(B)}
\]

\[
\Rightarrow\quad
\lambda_1(B)\ge \frac{c(d)^2}{2\Delta}\operatorname{diam}(B)^{-2}
\]

\textbf{Status:} Conditional on uniform isoperimetric bound.
